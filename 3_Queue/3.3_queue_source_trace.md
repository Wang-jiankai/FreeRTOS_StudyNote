## 队列核心结构

### xQUEUE 真实定义（queue.c）

```c:queue.c
typedef struct QueueDefinition
{
    int8_t * pcHead;              /* 队列存储区起始地址 */

    union {
        QueuePointers_t xQueue;        /* 队列模式用这些指针 */
        SemaphoreData_t xSemaphore;   /* 信号量/互斥量模式用这些字段 */
    } u;

    List_t xTasksWaitingToSend;   /* 等待发送（队列满）*/
    List_t xTasksWaitingToReceive;/* 等待接收（队列空）*/

    volatile UBaseType_t uxMessagesWaiting;  /* 当前消息数 */
    UBaseType_t uxLength;                    /* 队列深度 */
    UBaseType_t uxItemSize;                  /* 每个元素大小（字节）*/

    volatile int8_t cRxLock;   /* 接收锁定计数：中断中收到数据时累加，退出时统一处理 */
    volatile int8_t cTxLock;   /* 发送锁定计数：中断中发出数据时累加，退出时统一处理 */

    #if ( configUSE_QUEUE_SETS == 1 )
        struct QueueDefinition * pxQueueSetContainer;
    #endif

    #if ( configUSE_TRACE_FACILITY == 1 )
        UBaseType_t uxQueueNumber;
        uint8_t ucQueueType;
    #endif
} xQUEUE;

typedef xQUEUE Queue_t;
```

## xQueueSend（xQueueGenericSend）源码 trace


### 核心逻辑

```c:queue.c
BaseType_t xQueueGenericSend(QueueHandle_t xQueue,
                              const void * const pvItemToQueue,
                              TickType_t xTicksToWait,
                              const BaseType_t xCopyPosition)
{
    BaseType_t xEntryTimeSet = pdFALSE;
    TimeOut_t xTimeOut;

    for (;;) {
        taskENTER_CRITICAL();

        /* 检查队列是否有空间（uxMessagesWaiting < uxLength）*/
        if (uxQueueMessagesWaiting < pxQueue->uxLength) {
            /* 有空间，拷贝数据 */
            prvCopyDataToQueue(pxQueue, pvItemToQueue, xCopyPosition);

            /* 如果有任务在等接收，唤醒它 */
            if (listIS_EMPTY(&pxQueue->xTasksWaitingToReceive) == pdFALSE) {
                if (xTaskRemoveFromEventList(&pxQueue->xTasksWaitingToReceive) != pdFALSE) {
                    queueYIELD_IF_USING_PREEMPTION();  /* 触发 PendSV */
                }
            }

            taskEXIT_CRITICAL();
            return pdPASS;
        }

        /* 队列满，处理阻塞/超时 */
        if (xEntryTimeSet == pdFALSE) {
            vTaskSetTimeOutState(&xTimeOut);
            xEntryTimeSet = pdTRUE;
        }

        /* 当前任务加入等待发送链表（按优先级排序）*/
        vTaskPlaceOnUnorderedEventList(&pxQueue->xTasksWaitingToSend, xTicksToWait);
        taskEXIT_CRITICAL();
        portYIELD();  /* 让出 CPU */

        if (xTaskCheckTimeOut(&xTimeOut, &xTicksToWait) == pdTRUE) {
            taskEXIT_CRITICAL();
            return errQUEUE_FULL;
        }
        /* 循环重试 */
    }
}
```

### prvCopyDataToQueue（queue.c）

```c:queue.c
static void prvCopyDataToQueue(Queue_t * const pxQueue,
                               const void * pvItemToQueue,
                               const BaseType_t xCopyPosition)
{
    if (pxQueue->uxItemSize == 0) {
        return;  /* 信号量没有数据负载 */
    }

    if (xCopyPosition == queueSEND_TO_FRONT) {
        /* 发送到头部：移动读指针到最后一个位置 */
        pxQueue->u.xQueue.pcReadFrom = pxQueue->pcHead
            + (pxQueue->uxLength - 1) * pxQueue->uxItemSize;
    }

    /* 拷贝数据到写位置 */
    memcpy(pxQueue->u.xQueue.pcWriteTo, pvItemToQueue, pxQueue->uxItemSize);

    /* 更新写指针（循环）*/
    pxQueue->u.xQueue.pcWriteTo += pxQueue->uxItemSize;
    if (pxQueue->u.xQueue.pcWriteTo >= pxQueue->u.xQueue.pcTail) {
        pxQueue->u.xQueue.pcWriteTo = pxQueue->pcHead;
    }

    pxQueue->uxMessagesWaiting++;
}
```

## xQueueReceive 源码 trace

```c:queue.c
BaseType_t xQueueReceive(QueueHandle_t xQueue,
                          void * const pvBuffer,
                          TickType_t xTicksToWait)
{
    BaseType_t xEntryTimeSet = pdFALSE;
    TimeOut_t xTimeOut;

    for (;;) {
        taskENTER_CRITICAL();

        if (pxQueue->uxMessagesWaiting > 0) {
            /* 有数据，拷贝出来 */
            prvCopyDataFromQueue(pxQueue, pvBuffer);

            pxQueue->uxMessagesWaiting--;

            /* 唤醒一个等待发送的任务 */
            if (listIS_EMPTY(&pxQueue->xTasksWaitingToSend) == pdFALSE) {
                if (xTaskRemoveFromEventList(&pxQueue->xTasksWaitingToSend) != pdFALSE) {
                    queueYIELD_IF_USING_PREEMPTION();
                }
            }

            taskEXIT_CRITICAL();
            return pdPASS;
        }

        /* 队列空，处理阻塞/超时 */
        if (xEntryTimeSet == pdFALSE) {
            vTaskSetTimeOutState(&xTimeOut);
            xEntryTimeSet = pdTRUE;
        }

        vTaskPlaceOnUnorderedEventList(&pxQueue->xTasksWaitingToReceive, xTicksToWait);
        taskEXIT_CRITICAL();
        portYIELD();

        if (xTaskCheckTimeOut(&xTimeOut, &xTicksToWait) == pdTRUE) {
            taskEXIT_CRITICAL();
            return errQUEUE_EMPTY;
        }
    }
}
```

## cRxLock / cTxLock 机制详解

### 为什么需要锁定计数

中断中不能执行复杂的任务唤醒逻辑（可能触发调度），因此：

```c:queue.c
/* xQueueReceiveFromISR 简化版 */
BaseType_t xQueueReceiveFromISR(QueueHandle_t xQueue,
                                 void * const pvBuffer,
                                 BaseType_t * const pxHigherPriorityTaskWoken)
{
    /* 锁定计数 > 0 表示队列被锁定，中断中无法安全执行唤醒 */
    if (pxQueue->cRxLock == queueUNLOCKED) {
        /* 正常路径：直接处理数据 + 唤醒等待任务 */
        /* ... */
    } else {
        /* 锁定路径：只累加计数，不做复杂操作
         * 等中断退出后，在 PendSV 或 tick 中统一处理 */
        pxQueue->cRxLock++;
    }
}
```

**关键点**：
- 中断中操作队列时，`cRxLock` 或 `cTxLock` 累加
- 中断返回时检查锁定计数，如果有待处理操作就触发 PendSV 执行任务唤醒
- 这样保证了中断上下文的简洁性和可重入性

## 本节小结

- **xQUEUE** 使用 union 复用内存：`xQueue`（队列）和 `xSemaphore`（互斥量）共用同一块
- `cRxLock`/`cTxLock` 是**中断锁定计数器**，避免中断上下文做复杂唤醒操作
- `prvCopyDataToQueue` 处理循环指针更新和消息计数更新
- 发送/接收时如果对端有等待任务，通过 `xTaskRemoveFromEventList` 唤醒并可能触发调度
- 互斥量的 `xMutexHolder` 和 `uxRecursiveCallCount` 存在 `u.xSemaphore` union 成员中
